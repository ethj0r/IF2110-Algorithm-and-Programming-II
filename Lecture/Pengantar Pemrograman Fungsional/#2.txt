    coinConversion: integer, integer, integer, integer -> <integer, integer>
    {menerima input integer sejumlah koin quarter, dime, nickel, dan penny,
     lalu mengembalikan pasangan <dollar, sen> yang setara nilainya}

REALISASI
    coinConversion(quarter, dime, nickel, penny) :
                < ((25*quarter + 10*dime + 5*nickel + penny) div 100), ((25*quarter + 10*dime + 5*nickel + penny) mod 100) >

PENERAPAN
    coinConversion(8, 20, 30, 77) = <6, 27>
